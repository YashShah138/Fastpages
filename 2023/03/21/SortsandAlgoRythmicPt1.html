Bubble Sort 

public class BubbleSort {
    public static void bubbleSort(Comparable[] arr) {
        int n = arr.length;
        boolean swapped;
        do {
            swapped = false;
            for (int i = 1; i < n; i++) {
                if (arr[i-1].compareTo(arr[i]) > 0) {
                    Comparable temp = arr[i-1];
                    arr[i-1] = arr[i];
                    arr[i] = temp;
                    swapped = true;
                }
            }
            n--;
            System.out.println(Arrays.toString(arr));
        } while (swapped);
    }
    
    public static void main(String[] args) {
        Integer[] arr = { 5, 1, 4, 2, 8 };
        System.out.println("Before sorting: " + Arrays.toString(arr));
        bubbleSort(arr);
        System.out.println("After sorting: " + Arrays.toString(arr)); // [1, 2, 4, 5, 8]
    }
}

Selection Sort 

public class SelectionSort {
    public static void selectionSort(Comparable[] arr) {
        int n = arr.length;
        for (int i = 0; i < n-1; i++) {
            int minIdx = i;
            for (int j = i+1; j < n; j++) {
                if (arr[j].compareTo(arr[minIdx]) < 0) {
                    minIdx = j;
                }
                System.out.println(Arrays.toString(arr));
            }
            if (minIdx != i) {
                Comparable temp = arr[i];
                arr[i] = arr[minIdx];
                arr[minIdx] = temp;
            }
        }
    }
    
    public static void main(String[] args) {
        Integer[] arr = { 5, 1, 4, 2, 8 };
        System.out.println("Before sorting: " + Arrays.toString(arr));
        selectionSort(arr);
        System.out.println("After sorting: " + Arrays.toString(arr)); // [1, 2, 4, 5, 8]
    }
}

Insertion Sort 

public class InsertionSort {
    public static void insertionSort(Comparable[] arr) {
        int n = arr.length;
        for (int i = 1; i < n; i++) {
            Comparable key = arr[i];
            int j = i - 1;
            while (j >= 0 && arr[j].compareTo(key) > 0) {
                arr[j+1] = arr[j];
                j--;
                System.out.println(Arrays.toString(arr));
            }
            arr[j+1] = key;
            System.out.println(Arrays.toString(arr));
        }
    }
    
    public static void main(String[] args) {
        Integer[] arr = { 5, 1, 4, 2, 8 };
        System.out.println("Before sorting: " + Arrays.toString(arr));
        insertionSort(arr);
        System.out.println("After sorting: " + Arrays.toString(arr)); // [1, 2, 4, 5, 8]
    }
}

Merge Sort 

public class MergeSort {

    public static void mergeSort(Comparable[] arr, int l, int r) {
        if (l < r) {
            int m = (l + r) / 2;
            mergeSort(arr, l, m);
            mergeSort(arr, m+1, r);
            merge(arr, l, m, r);
        }
    }

    public static void merge(Comparable[] arr, int l, int m, int r) {
        int n1 = m - l + 1;
        int n2 = r - m;
        Comparable[] L = new Comparable[n1];
        Comparable[] R = new Comparable[n2];
        for (int i = 0; i < n1; i++) {
            L[i] = arr[l+i];
        }
        for (int j = 0; j < n2; j++) {
            R[j] = arr[m+1+j];
        }
        int i = 0, j = 0;
        int k = l;
        while (i < n1 && j < n2) {
            if (L[i].compareTo(R[j]) <= 0) {
                arr[k] = L[i];
                i++;
            } else {
                arr[k] = R[j];
                j++;
            }
            k++;
        }
        System.out.println(Arrays.toString(arr));
        while (i < n1) {
            arr[k] = L[i];
            i++;
            k++;
        }
        System.out.println(Arrays.toString(arr));
        while (j < n2) {
            arr[k] = R[j];
            j++;
            k++;
        }
        System.out.println(Arrays.toString(arr));
    }

    public static void main(String[] args) {
        Integer[] arr = {5, 2, 8, 3, 1, 6};
        System.out.println("Before sorting: " + Arrays.toString(arr));
        mergeSort(arr, 0, arr.length - 1);
        System.out.println("After sorting: " + Arrays.toString(arr));
    }
    
}
